// Fill these in from your Supabase project:
// Project Settings → API → Project URL / anon public key.
// The anon key is safe to expose in client-side code — it only works
// within the limits of the Row Level Security policies in sql/schema.sql.
const SUPABASE_URL = 'https://YOUR-PROJECT-REF.supabase.co';
const SUPABASE_ANON_KEY = 'YOUR-ANON-PUBLIC-KEY';

const supabaseClient = supabase.createClient(SUPABASE_URL, SUPABASE_ANON_KEY);

const STATUS_META = {
  created: { label: 'Created', color: '#5B6470' },
  dispatched: { label: 'Dispatched', color: '#3F5B73' },
  in_transit: { label: 'In Transit', color: '#C98A2B' },
  out_for_delivery: { label: 'Out For Delivery', color: '#C15B2C' },
  delivered: { label: 'Delivered', color: '#2F7A4D' },
  exception: { label: 'Exception', color: '#B5452F' },
};

function statusLabel(status) {
  return (STATUS_META[status] && STATUS_META[status].label) || status;
}

function statusColor(status) {
  return (STATUS_META[status] && STATUS_META[status].color) || '#5B6470';
}

// Renders a status as a row of "split-flap" character tiles inside `el`
// and triggers the flip-in animation. Used for the one or two hero status
// displays on a page — not for compact rows in a table.
function renderStatusBoard(el, status) {
  const label = statusLabel(status).toUpperCase();
  el.innerHTML = '';
  [...label].forEach((ch, i) => {
    const span = document.createElement('span');
    span.className = 'flap';
    span.textContent = ch === ' ' ? '\u00A0' : ch;
    span.style.animationDelay = `${i * 25}ms`;
    el.appendChild(span);
  });
  requestAnimationFrame(() => {
    el.querySelectorAll('.flap').forEach((s) => s.classList.add('flip'));
  });
}

let currentParcel = null;
let pinMap = null;
let pinMarker = null;

function show(viewId) {
  ['view-list', 'view-new', 'view-detail'].forEach((id) => {
    document.getElementById(id).classList.toggle('hidden', id !== viewId);
  });
}

document.querySelectorAll('.back-btn').forEach((btn) => {
  btn.addEventListener('click', () => show(btn.dataset.back));
});

function fmtDate(d) {
  return d ? new Date(d).toLocaleString('en-GB') : '—';
}

// ---------- List ----------
async function loadParcelList() {
  const tbody = document.getElementById('parcel-table-body');
  const emptyEl = document.getElementById('list-empty');
  tbody.innerHTML = '';

  const { data, error } = await supabaseClient
    .from('parcels')
    .select('*')
    .order('updated_at', { ascending: false });

  if (error) {
    console.error(error);
    return;
  }

  emptyEl.classList.toggle('hidden', data.length > 0);

  data.forEach((p) => {
    const tr = document.createElement('tr');
    tr.innerHTML = `
      <td>${p.tracking_number}</td>
      <td>${p.recipient_name}</td>
      <td>${p.destination}</td>
      <td><span class="status-pill" style="background:${statusColor(p.status)}">${statusLabel(p.status)}</span></td>
      <td>${fmtDate(p.updated_at)}</td>
      <td><button class="btn-secondary view-btn" data-id="${p.id}">View</button></td>
    `;
    tbody.appendChild(tr);
  });

  tbody.querySelectorAll('.view-btn').forEach((btn) => {
    btn.addEventListener('click', () => openDetail(btn.dataset.id));
  });
}

document.getElementById('new-parcel-btn').addEventListener('click', () => {

function generateTrackingNumber() {
  const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789'; // no O/0/I/1, easy to misread
  let code = 'UK';
  for (let i = 0; i < 10; i++) code += chars[Math.floor(Math.random() * chars.length)];
  return code;
}

document.getElementById('new-parcel-form').addEventListener('submit', async (e) => {
  e.preventDefault();
  const errorEl = document.getElementById('new-parcel-error');
  errorEl.classList.add('hidden');

  const recipient_name = document.getElementById('np-recipient').value.trim();
  const origin = document.getElementById('np-origin').value.trim();
  const destination = document.getElementById('np-destination').value.trim();
  const estimated_delivery = document.getElementById('np-eta').value || null;

  let attempts = 0;
  let created = null;
  let lastError = null;

  while (attempts < 5 && !created) {
    attempts++;
    const tracking_number = generateTrackingNumber();
    const { data, error } = await supabaseClient
      .from('parcels')
      .insert({ tracking_number, recipient_name, origin, destination, estimated_delivery })
      .select()
      .single();

    if (!error) {
      created = data;
    } else if (error.code !== '23505') { // not a duplicate-tracking-number retry case
      lastError = error;
      break;
    }
  }

  if (!created) {
    errorEl.textContent = lastError ? lastError.message : 'Could not generate a unique tracking number — please try again.';
    errorEl.classList.remove('hidden');
    return;
  }

  await loadParcelList();
  openDetail(created.id);
});

// ---------- Detail ----------
async function openDetail(parcelId) {
  const { data: parcel, error } = await supabaseClient
    .from('parcels')
    .select('*')
    .eq('id', parcelId)
    .single();

  if (error || !parcel) {
    console.error(error);
    show('view-list');
    return;
  }

  currentParcel = parcel;
  document.getElementById('detail-title').textContent = `${parcel.tracking_number} — ${parcel.recipient_name}`;
  document.getElementById('detail-origin').textContent = parcel.origin;
  document.getElementById('detail-destination').textContent = parcel.destination;
  renderStatusBoard(document.getElementById('detail-status-board'), parcel.status);

  await loadDetailTimeline(parcelId);

  document.getElementById('add-event-form').reset();
  document.getElementById('ev-status').value = parcel.status;
  document.getElementById('add-event-error').classList.add('hidden');

  show('view-detail');

  // Leaflet needs a visible container to size itself correctly, so
  // (re)initialise / invalidate on the next tick, after 'hidden' is removed.
  setTimeout(() => {
    if (!pinMap) {
      pinMap = L.map('pin-map').setView([54.5, -3.5], 6);
      L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
        attribution: '&copy; OpenStreetMap contributors',
        maxZoom: 18,
      }).addTo(pinMap);
      pinMap.on('click', (e) => setPin(e.latlng.lat, e.latlng.lng));
    }
    if (pinMarker) {
      pinMap.removeLayer(pinMarker);
      pinMarker = null;
    }
    pinMap.invalidateSize();
  }, 0);
}

async function loadDetailTimeline(parcelId) {
  const timelineEl = document.getElementById('detail-timeline');
  timelineEl.innerHTML = '';

  const { data: events, error } = await supabaseClient
    .from('tracking_events')
    .select('*')
    .eq('parcel_id', parcelId)
    .order('created_at', { ascending: false });

  if (error) {
    console.error(error);
    return;
  }

  events.forEach((e) => {
    const li = document.createElement('li');
    li.className = 'timeline-item';
    li.innerHTML = `
      <span class="timeline-dot" style="background:${statusColor(e.status)}"></span>
      <div class="timeline-content">
        <div class="timeline-status">${statusLabel(e.status)}</div>
        <div class="timeline-location">${e.location_name}${e.note ? ' — ' + e.note : ''}</div>
        <div class="timeline-time">${fmtDate(e.created_at)}</div>
      </div>
    `;
    timelineEl.appendChild(li);
  });
}

function setPin(lat, lng) {
  document.getElementById('ev-lat').value = lat.toFixed(6);
  document.getElementById('ev-lng').value = lng.toFixed(6);
  if (pinMarker) {
    pinMarker.setLatLng([lat, lng]);
  } else {
    pinMarker = L.marker([lat, lng], { draggable: true }).addTo(pinMap);
    pinMarker.on('dragend', () => {
      const pos = pinMarker.getLatLng();
      document.getElementById('ev-lat').value = pos.lat.toFixed(6);
      document.getElementById('ev-lng').value = pos.lng.toFixed(6);
    });
  }
  pinMap.setView([lat, lng], 11);
}

document.getElementById('geocode-btn').addEventListener('click', async () => {
  const query = document.getElementById('ev-location').value.trim();
  if (!query) return;
  const errorEl = document.getElementById('add-event-error');
  errorEl.classList.add('hidden');

  try {
    const url = `https://nominatim.openstreetmap.org/search?format=json&limit=1&countrycodes=gb&q=${encodeURIComponent(query)}`;
    const res = await fetch(url, { headers: { 'Accept-Language': 'en-GB' } });
    const results = await res.json();
    if (!results.length) {
      errorEl.textContent = 'No matching UK location found — you can drop a pin on the map instead.';
      errorEl.classList.remove('hidden');
      return;
    }
    setPin(parseFloat(results[0].lat), parseFloat(results[0].lon));
  } catch (err) {
    console.error(err);
    errorEl.textContent = 'Location lookup failed — you can drop a pin on the map instead.';
    errorEl.classList.remove('hidden');
  }
});

document.getElementById('add-event-form').addEventListener('submit', async (e) => {
  e.preventDefault();
  const errorEl = document.getElementById('add-event-error');
  errorEl.classList.add('hidden');

  const status = document.getElementById('ev-status').value;
  const location_name = document.getElementById('ev-location').value.trim();
  const latitude = parseFloat(document.getElementById('ev-lat').value);
  const longitude = parseFloat(document.getElementById('ev-lng').value);
  const note = document.getElementById('ev-note').value.trim() || null;

  if (Number.isNaN(latitude) || Number.isNaN(longitude)) {
    errorEl.textContent = 'Set a location using "Find" or by clicking the map first.';
    errorEl.classList.remove('hidden');
    return;
  }

  const { data: { session } } = await supabaseClient.auth.getSession();

  const { error } = await supabaseClient.from('tracking_events').insert({
    parcel_id: currentParcel.id,
    status,
    location_name,
    latitude,
    longitude,
    note,
    created_by: session?.user?.id || null,
  });

  if (error) {
    errorEl.textContent = error.message;
    errorEl.classList.remove('hidden');
    return;
  }

  await openDetail(currentParcel.id);
  await loadParcelList();
});

// ---------- Init ----------
async function init() {
  const { data: { session } } = await supabaseClient.auth.getSession();
  if (!session) {
    window.location.href = 'staff-login.html';
    return;
  }
  document.getElementById('staff-email').textContent = session.user.email;

  document.getElementById('sign-out').addEventListener('click', async () => {
    await supabaseClient.auth.signOut();
    window.location.href = 'staff-login.html';
  });

  await loadParcelList();
  show('view-list');
}

init();
